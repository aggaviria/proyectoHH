    <script>
        /* Módulo 1 → Estado: variables globales (apiKey, conversationHistory) */
        let apiKey = '';
        let conversationHistory = [];

        /* Módulo 2 → RAG: base de conocimiento (vacío por ahora) */
        let KNOWLEDGE_BASE = "";

        async function loadKnowledgeBase() {
            try {
                const response = await fetch('./catalogo.txt');
                if (!response.ok) throw new Error("No se pudo cargar el archivo catalogo.txt");
                KNOWLEDGE_BASE = await response.text();
                console.log("Base de conocimiento cargada con éxito.");
                buildSystemPrompt(); // Construir el prompt una vez cargados los datos
            } catch (error) {
                console.error("Error cargando RAG:", error);
                addMessage('assistant', "Error crítico: No se pudo cargar el catálogo de conocimientos.");
            }
        }

        // Ejecutar carga al iniciar
        window.onload = loadKnowledgeBase;

        /* Módulo 3 → System Prompt: reglas del asistente (vacío por ahora) */
        let SYSTEM_PROMPT = "";

        function buildSystemPrompt() {
            SYSTEM_PROMPT = `Eres el asistente de NovaSupport.

REGLAS CRÍTICAS:
1. Responde ÚNICAMENTE usando la información del catálogo proporcionado.
2. Si la información no está en el catálogo, di: "Esa información no está disponible. Te recomiendo contactar directamente con NovaSupport."
3. Tono: profesional y conciso.
4. Nunca inventes datos que no estén en el catálogo.

BASE DE CONOCIMIENTO:
${KNOWLEDGE_BASE}`;
        }

        /* Módulo 4 → API Key: función saveApiKey() */
        function saveApiKey() {
            const input = document.getElementById('apiKey');
            const btn = document.querySelector('header button');
            const value = input.value.trim();

            if (value.startsWith('sk-')) {
                apiKey = value;
                input.disabled = true;
                btn.innerHTML = '✓ Conectado';
                btn.style.background = '#10B981'; // Cambio visual a verde éxito
                btn.onclick = null;
                console.log("API Key conectada con éxito.");
            } else {
                alert("La clave debe comenzar con 'sk-'. Por favor, verifica tu API Key.");
            }
        }

        /* Módulo 5 → Enviar mensaje: función sendMessage() */
        async function sendMessage() {
            const textarea = document.getElementById('userInput');
            const text = textarea.value.trim();

            if (!text) return;
            if (!apiKey) {
                alert("Por favor, ingresa y conecta tu API Key primero.");
                return;
            }

            // Limpiar UI y actualizar historial
            textarea.value = '';
            autoResize(textarea);
            addMessage('user', text);
            conversationHistory.push({ role: 'user', content: text });

            showTyping(true);

            try {
                const response = await fetch('https://api.openai.com/v1/chat/completions', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${apiKey}`
                    },
                    body: JSON.stringify({
                        model: 'gpt-4o-mini',
                        temperature: 0.3,
                        max_tokens: 600,
                        messages: [{ role: 'system', content: SYSTEM_PROMPT }, ...conversationHistory]
                    })
                });

                const data = await response.json();
                
                if (data.error) throw new Error(data.error.message);

                const aiText = data.choices[0].message.content;
                
                showTyping(false);
                addMessage('assistant', aiText);
                conversationHistory.push({ role: 'assistant', content: aiText });

            } catch (error) {
                showTyping(false);
                addMessage('assistant', "Lo siento, hubo un error: " + error.message);
                console.error("OpenAI API Error:", error);
            }
        }

        /* Módulo 6 → UI helpers: funciones de interfaz */
        function addMessage(role, text) {
            const chatWindow = document.getElementById('chatWindow');
            const msgDiv = document.createElement('div');
            
            // Asignar clase según rol (ia para assistant, user para usuario)
            msgDiv.className = `msg-placeholder ${role === 'assistant' ? 'ia' : 'user'}`;
            msgDiv.innerHTML = escapeHtml(text);
            
            chatWindow.appendChild(msgDiv);
            chatWindow.scrollTop = chatWindow.scrollHeight;
        }

        function showTyping(visible) {
            const chatWindow = document.getElementById('chatWindow');
            const typingId = 'typing-indicator';
            let indicator = document.getElementById(typingId);

            if (visible && !indicator) {
                indicator = document.createElement('div');
                indicator.id = typingId;
                indicator.className = 'msg-placeholder ia';
                indicator.style.opacity = '0.6';
                indicator.innerHTML = '<em>Escribiendo...</em>';
                chatWindow.appendChild(indicator);
                chatWindow.scrollTop = chatWindow.scrollHeight;
            } else if (!visible && indicator) {
                indicator.remove();
            }
        }

        function autoResize(el) {
            el.style.height = 'auto';
            el.style.height = (el.scrollHeight) + 'px';
        }

        function handleKey(e) {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMessage();
            }
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
